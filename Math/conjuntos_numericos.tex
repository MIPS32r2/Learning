\section{Conjuntos numéricos}
  \begin{enumerate}
    \item Propriedades da Soma

    S1) (Associativa) $a + (b + c) = (a + b) + c$

    S2) (Comunicativa) $a + b = b +a$

    S3) (Elemento Neutro) $a + 0 = a$ %O zero é o elemento neutro da soma de números naturais

  \end{enumerate}

\begin{enumerate} 
  \item Propriedades da Multiplicação
  
    M1) (Associativa) $a(bc) = (ab)c$

    M2) (Comutativa) $ab = ba$ % "A ordem dos fatores não altera o produto"

    M3) (Elemento Neutro) $1a = a$

  \end{enumerate}
